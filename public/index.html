-- ============================================ -- FITAI PRO V3 — PRODUCTION SQL -- ============================================ -- Exécuter dans Supabase SQL Editor -- ============================================ -- TABLE: public_profiles -- ============================================ CREATE TABLE IF NOT EXISTS public_profiles ( user_id UUID PRIMARY KEY REFERENCES auth.users(id) ON DELETE CASCADE, display_name TEXT CHECK (char_length(display_name) <= 40), created_at TIMESTAMPTZ NOT NULL DEFAULT NOW(), updated_at TIMESTAMPTZ NOT NULL DEFAULT NOW() ); CREATE INDEX IF NOT EXISTS idx_public_profiles_display_name ON public_profiles(display_name); ALTER TABLE public_profiles ENABLE ROW LEVEL SECURITY; DROP POLICY IF EXISTS "public_profiles_select_all" ON public_profiles; CREATE POLICY "public_profiles_select_all" ON public_profiles FOR SELECT USING (true); DROP POLICY IF EXISTS "public_profiles_insert_own" ON public_profiles; CREATE POLICY "public_profiles_insert_own" ON public_profiles FOR INSERT WITH CHECK (auth.uid() = user_id); DROP POLICY IF EXISTS "public_profiles_update_own" ON public_profiles; CREATE POLICY "public_profiles_update_own" ON public_profiles FOR UPDATE USING (auth.uid() = user_id) WITH CHECK (auth.uid() = user_id); -- ============================================ -- TABLE: workouts -- ============================================ CREATE TABLE IF NOT EXISTS workouts ( id UUID PRIMARY KEY DEFAULT gen_random_uuid(), user_id UUID NOT NULL REFERENCES auth.users(id) ON DELETE CASCADE, is_public BOOLEAN NOT NULL DEFAULT true, title TEXT NOT NULL DEFAULT 'Workout' CHECK (char_length(title) <= 120), intensity TEXT CHECK (intensity IN ('easy', 'medium', 'hard', 'killer')), notes TEXT CHECK (char_length(notes) <= 500), plan_json JSONB, created_at TIMESTAMPTZ NOT NULL DEFAULT NOW() ); CREATE INDEX IF NOT EXISTS idx_workouts_user_created ON workouts(user_id, created_at DESC); CREATE INDEX IF NOT EXISTS idx_workouts_public_created ON workouts(is_public, created_at DESC) WHERE is_public = true; CREATE INDEX IF NOT EXISTS idx_workouts_user_public ON workouts(user_id, is_public); ALTER TABLE workouts ENABLE ROW LEVEL SECURITY; DROP POLICY IF EXISTS "workouts_select_public_or_own" ON workouts; CREATE POLICY "workouts_select_public_or_own" ON workouts FOR SELECT USING (is_public = true OR auth.uid() = user_id); DROP POLICY IF EXISTS "workouts_insert_own" ON workouts; CREATE POLICY "workouts_insert_own" ON workouts FOR INSERT WITH CHECK (auth.uid() = user_id); DROP POLICY IF EXISTS "workouts_update_own" ON workouts; CREATE POLICY "workouts_update_own" ON workouts FOR UPDATE USING (auth.uid() = user_id) WITH CHECK (auth.uid() = user_id); DROP POLICY IF EXISTS "workouts_delete_own" ON workouts; CREATE POLICY "workouts_delete_own" ON workouts FOR DELETE USING (auth.uid() = user_id); -- ============================================ -- TABLE: kudos -- ============================================ CREATE TABLE IF NOT EXISTS kudos ( id UUID PRIMARY KEY DEFAULT gen_random_uuid(), workout_id UUID NOT NULL REFERENCES workouts(id) ON DELETE CASCADE, user_id UUID NOT NULL REFERENCES auth.users(id) ON DELETE CASCADE, created_at TIMESTAMPTZ NOT NULL DEFAULT NOW(), UNIQUE(workout_id, user_id) ); CREATE INDEX IF NOT EXISTS idx_kudos_workout ON kudos(workout_id); CREATE INDEX IF NOT EXISTS idx_kudos_user ON kudos(user_id); ALTER TABLE kudos ENABLE ROW LEVEL SECURITY; DROP POLICY IF EXISTS "kudos_select_all" ON kudos; CREATE POLICY "kudos_select_all" ON kudos FOR SELECT USING (true); DROP POLICY IF EXISTS "kudos_insert_own" ON kudos; CREATE POLICY "kudos_insert_own" ON kudos FOR INSERT WITH CHECK (auth.uid() = user_id); DROP POLICY IF EXISTS "kudos_delete_own" ON kudos; CREATE POLICY "kudos_delete_own" ON kudos FOR DELETE USING (auth.uid() = user_id); -- ============================================ -- TABLE: body_scans -- ============================================ CREATE TABLE IF NOT EXISTS body_scans ( id UUID PRIMARY KEY DEFAULT gen_random_uuid(), user_id UUID NOT NULL REFERENCES auth.users(id) ON DELETE CASCADE, file_path TEXT NOT NULL, meta JSONB, created_at TIMESTAMPTZ NOT NULL DEFAULT NOW() ); CREATE INDEX IF NOT EXISTS idx_body_scans_user_created ON body_scans(user_id, created_at DESC); ALTER TABLE body_scans ENABLE ROW LEVEL SECURITY; DROP POLICY IF EXISTS "body_scans_select_own" ON body_scans; CREATE POLICY "body_scans_select_own" ON body_scans FOR SELECT USING (auth.uid() = user_id); DROP POLICY IF EXISTS "body_scans_insert_own" ON body_scans; CREATE POLICY "body_scans_insert_own" ON body_scans FOR INSERT WITH CHECK (auth.uid() = user_id); DROP POLICY IF EXISTS "body_scans_delete_own" ON body_scans; CREATE POLICY "body_scans_delete_own" ON body_scans FOR DELETE USING (auth.uid() = user_id); -- ============================================ -- TABLE: reports (abuse reporting) -- ============================================ CREATE TABLE IF NOT EXISTS reports ( id UUID PRIMARY KEY DEFAULT gen_random_uuid(), workout_id UUID NOT NULL REFERENCES workouts(id) ON DELETE CASCADE, reporter_user_id UUID NOT NULL REFERENCES auth.users(id) ON DELETE CASCADE, reason TEXT CHECK (char_length(reason) <= 200), created_at TIMESTAMPTZ NOT NULL DEFAULT NOW(), UNIQUE(workout_id, reporter_user_id) ); CREATE INDEX IF NOT EXISTS idx_reports_workout ON reports(workout_id); CREATE INDEX IF NOT EXISTS idx_reports_created ON reports(created_at DESC); ALTER TABLE reports ENABLE ROW LEVEL SECURITY; DROP POLICY IF EXISTS "reports_insert_authenticated" ON reports; CREATE POLICY "reports_insert_authenticated" ON reports FOR INSERT WITH CHECK (auth.uid() = reporter_user_id); -- Admin peut lire tous les reports (ajuster selon besoins) DROP POLICY IF EXISTS "reports_select_own" ON reports; CREATE POLICY "reports_select_own" ON reports FOR SELECT USING (auth.uid() = reporter_user_id); -- ============================================ -- VIEW: workouts_feed -- ============================================ DROP VIEW IF EXISTS workouts_feed; CREATE VIEW workouts_feed AS SELECT w.id, w.user_id, w.is_public, w.title, w.intensity, w.notes, w.plan_json, w.created_at, COALESCE(p.display_name, 'Anonymous') AS display_name, (SELECT COUNT(*) FROM kudos k WHERE k.workout_id = w.id) AS kudos_count, EXISTS( SELECT 1 FROM kudos k2 WHERE k2.workout_id = w.id AND k2.user_id = auth.uid() ) AS liked_by_me FROM workouts w LEFT JOIN public_profiles p ON p.user_id = w.user_id WHERE w.is_public = true OR w.user_id = auth.uid();
